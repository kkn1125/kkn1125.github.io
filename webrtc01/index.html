node.js express + React 환경 구축부터 해보자

이번에 WebRTC와 webSocket을 조사해야할 일이 생겨서 부랴부랴 이론을 보고, 테스트해보기 위해 여러 블로그와 문서를 찾아다녔습니다. 이미 어려운걸 지금부터 해야한다니... 복 받은건가

먼저 Next에서 진행하려고 했으나 WebRTC를 설정하는데 있어서 node.js도 공부해 볼 겸 node와 react를 결합한 환경에서 테스트 해보려합니다.

webrtc_test라는 폴더를 만들고 이동합니다.

1mkdir webrtc_test && cd "$_"

그리고 client디렉토리를 먼저 만들어 줄 텐데요. react프로젝트를 만듭니다.

1npm install -g create-react-app # create-react-app이 없다면 설치해주세요!
2npx create-react-app client

node.js express WebSocket 설정

client에서는 굳이 설치할 종속성은 없으니 바로 node도 세팅해줍니다. webrtc_test를 루트로 잡고 아래 명령줄을 이어서 입력합니다.

1npm install express

참고로 파일구조는 아래와 같이 생성되어야 합니다.

express가 설치되면 server디렉토리를 생성하고 그 안에 server.js를 만들어 줍니다.

1mkdir server/ && cd "$_" && touch server.js

app과 socketServer를 따로 만듭니다. 이게 맞는 방법인지는 모르겠습니다만 작동이 되는 점에서 일단 이대로 진행을 했습니다. 추후 더 알아보고 수정하려합니다.

1// server.js
2
3const express = require("express");
4const app = express();
5const socketServer = express();
6const index = require("./Router/index");
7
8app.use("/", index);
9
10const appport = 4000;
11app.listen(appport, () => console.log(4000));
12
13const port = 5000;
14const HTTPServer = socketServer.listen(port, () => console.log(`${port}`));
15
16// websocket 서버를 위한 ws모듈 생성합니다.
17const { WebSocketServer } = require("ws");
18const webSocketServer = new WebSocketServer({
19  server: HTTPServer, // WebSocket서버에 연결할 HTTP서버를 지정한다.
20  // port: 30002, // WebSocket연결에 사용할 port를 지정한다(생략시, http서버와 동일한 port 공유 사용)
21  // noServer: true
22  // 위 속성들은 셋 중 하나만 설정해야합니다.
23});
24
25// or
26
27const wsModule = require("ws");
28const webSocketServer = wsModule.Server({
29  server: HTTPServer, // WebSocket서버에 연결할 HTTP서버를 지정한다.
30  // port: 30002, // WebSocket연결에 사용할 port를 지정한다(생략시, http서버와 동일한 port 공유 사용)
31  // noServer: true
32  // 위 속성들은 셋 중 하나만 설정해야합니다.
33});

그 다음으로 이벤트 처리하는 구문을 작성합니다.

1// server.js
2// 위에 썼던 코드들...
3
4webSocketServer.on("connection", (ws, request) => {
5  console.log(`클라이언트 접속`);
6
7  // 클라이언트로 메시지 전송
8  if (ws.readyState === ws.OPEN) {
9    ws.send(`서버 접속 완료`); // 데이터 전송
10  }
11
12  // 클라이언트에서 온 메시지 수신 이벤트 처리
13  ws.on("message", (msg) => {
14    console.log(`수신한 메시지 : ${msg}`);
15    ws.send("수신 완료");
16  });
17
18  // 에러 이벤트 처리
19  ws.on("error", (e) => {
20    console.log(`연결 에러발생 : ${e}`);
21  });
22
23  // 연결 종료 이벤트 처리
24  ws.on("close", () => {
25    console.log(`연결 종료`);
26  });
27});

파일구조를 참고하시면 server.js가 위치한 곳에 router 디렉토리를 만들고 그 안에 index.js를 만듭니다. node에서 사용할 페이지입니다.

1mkdir router/ && "$_"
2touch index.js

그 다음 아래와 같이 아무 내용이나 적어둡니다. 단지 url 연결이 잘 되는지 보기위함 입니다.

1const express = require("express");

3
4router.get("/", function (req, res) {
5  res.send(

이제 서버 쪽은 끝났습니다.

React WebSocket + WebRTC 설정

클라이언트에서 WebSocket 관련해서는 받을게 없습니다. 다 내장되어 있는 기능들이기 때문입니다.

먼저 불필요한 css와 App.js의 내용을 모두 날려버리고 작성합니다.

1import React, { useEffect, useRef, useState } from "react";
2
3const webSocket = new WebSocket("ws://localhost:5000");
4
5// 연결 이벤트
6webSocket.onopen = () => {
7  alert(`서버 연결!`);
8};
9
10// 메세지 이벤트
11webSocket.onmessage = function (e) {
12  console.log(`Server >> ${e.data}`);
13};
14
15// 종료 이벤트
16webSocket.onclose = function () {
17  console.log("서버 연결 종료");
18};
19
20// 에러 이벤트
21webSocket.onerror = function (e) {
22  console.log(e);
23};
24
25function App() {
26  const videoRef = useRef();
27  const [isConnect, setIsConnect] = useState(null);
28  const [value, setValue] = useState("");
29
30  const handleSubmit = (e) => {
31    e.preventDefault();
32    webSocket.send(value);
33    setValue("");
34    return false;
35  };
36
37  const handleValue = (e) => {
38    setValue(e.target.value);
39  };
40
41  const handleClose = function () {
42    if (webSocket.readyState === webSocket.OPEN) {
43      webSocket.close(); // 서버 연결 종료
44    } else {
45      alert("연결된 서버가 없습니다.");
46    }
47  };
48
49  useEffect(() => {
50    if (webSocket.readyState === webSocket.OPEN) {
51      setIsConnect(true);
52    }
53    var constraints = {
54      video: true,
55    };
56
57    var video = videoRef.current;
58
59    function handleSuccess(stream) {
60      window.stream = stream;
61      video.srcObject = stream;
62    }
63
64    function handleError(error) {
65      console.log("getUserMedia error: ", error);
66    }
67
68    window.navigator.mediaDevices
69      ?.getUserMedia(constraints)
70      .then((d) => {
71        return handleSuccess(d);
72      })
73      .catch((d) => {
74        return handleError(d);
75      });
76  }, []);
77
78  return (
79    <div>
80      <div>
81        {isConnect ? <span>연결 되었습니다.</span> : <span>연결 중...</span>}
82      </div>
83      <form onSubmit={handleSubmit}>
84        {isConnect && (
85          <input
86            type='text'
87            placeholder='메세지 입력'
88            value={value}
89            onChange={handleValue}
90          />
91        )}
92      </form>
93      <video ref={videoRef} autoPlay playsInline></video>
94      <button id='btn_close' onClick={handleClose}>
95        close
96      </button>
97    </div>
98  );
99}

이제 RTC 설정도 끝났으니 잘 작동되는지 봐야합니다. 특히나 카메라 권한을 요청하고 정상적으로 video에 카메라를 연결해주는지를요.

여기서 중요한 점이 있습니다. react를 그냥 실행하면 http 프로토콜로 실행 됩니다. 카메라 권한은 https 프로토콜로 보안연결이 되어야 카메라 권한 요청이 됩니다. http는 보안이 취약해서 카메라 권한 요청이 제한되어 아무런 반응을 하지 않습니다!

이것 때문에 3시간을 버렸습니다... 🥲

react로 https를 말하는 이유는 클라이언트에서 WebRTC를 설정했기 때문에 react를 https로 실행해야한다는 말입니다. 참고로 node에서 https로 실행하는 방법은 많습니다.

  1. mkcert를 설치해서 가짜 인증서를 받고 실행한다.
  2. letsencrypt-win-simple로 인증서 받고 실행한다.
  3. node의 https.createServer를 사용한다.
  4. 등등

하지만 react를 https로 하려면 어떻게 해야할까요?

간단하지만 보안 취약성을 염두에 두고 문제가 없다면 사용하시기 바랍니다. mac에 대해서 https를 설정하는 방법은 찾았지만 정작 저는 window라 이것저것 시도를 해봤습니다.

어렵지않게 알 수 있었는데요, 아래와 같이 client디렉토리에서 .env파일을 생성하고 HTTPS속성을 추가만해주면 https로 실행됩니다.

1HTTPS=true

2
3# 필요시

4HOST=0.0.0.0

5PORT=3000

이렇게 설정하고 그냥 실행하면 됩니다. 물론 node켜고, react키는게 불편하겠죠. 그래서 concurrently를 설치해서 express가 설치된 package.json을 좀 만져줍니다.

먼저 필요한 종속성을 설치합니다.

1npm install nodemon concurrently --save

저는 yarn을 사용하기 때문에 다소 다를 수 있는 부분이 있습니다. npm을 사용하신다면 그에 맞게 명령줄을 수정해주세요

그리고 package.json을 아래와 같이 마저 수정합니다.

1{
2  // ...
3
4  "scripts": {
5    "server": "cd server && nodemon server.js",
6    "client": "cd client && npm start",
7    "start": "concurrently --kill-others-on-fail \"yarn server\" \"yarn client\""
8  },
9  
10  // ...
11}

yarn start 해보면 3000, 4000, 5000 포트가 열리고 접속해서 테스트 해봅니다.

3000은 react, 4000은 express index페이지, 5000은 웹소켓 서버입니다.

먼저 실행된 react를 보면 아래와 같이 뜨면 성공입니다! 이미 여러 번 시도한 저는 카메라 권한 요청이 다시 뜨지 않지만 처음 시작할 때는 요청이 올 것 입니다.

콘솔에서는 웹소켓이 연결되었고, node console에서도 연결되었다고 뜨면 대성공입니다! 😎

영상이 반대로 나옵니다. 따로 설정하시려면 반대로 뒤집으시길...
