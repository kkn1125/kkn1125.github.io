종류도 많으니 필요하신 분은 들어가셔서 참고하시면 되겠습니다.

15let request = {
16    // 자바와 달리 파라미터값 조회 기능이 없어
17    // 만들어줍니다.
18    window: window,
19    // window프로퍼티에 전역객체 window를 저장합니다.
20    // 객체함수의 함수 내에서 this가 객체 자신이 되어서 window가 잡히지 않습니다.
21    getParameter: function(param){
22        // 흔히 보던 함수명
23        if(typeof param == 'string'){
24            let map = new Map();
25            let params = this.window.location.search.slice(1).split("&");
26            // search에서 ?기호를 잘라내고 이후 파라미터 증가 대비하여 &기호로 나눕니다.
27            for(let key of params){
28                let sep = key.split("=");
29                map.set(sep[0],sep[1]);
30                // =기호로 다시 나누어 map에 key-value로 담습니다.
31            }
32            if(map.get(param)==undefined){
33                // 타입 판별하기위해 간단히 넣었습니다.
34                throw "파라미터가 존재하지 않습니다.";
35            } else {
36                return map.get(param);
37                // 완료되면 map을 반환합니다.
38            }
39        } else {
40            throw "형식이 잘못되었습니다.";
41        }
42    }
43}

그나마 보기 좋게 객체형태로 나왔습니다. 여기서 이 객체를 필요한 정보만 골라내어 따로 Festv라는 객체를 만들어 리스트를 만들어 뿌리도록 하겠습니다.

1/**
2 * 공연 정보 객체
3 * @param {int} num 번호
4 * @param {string} subject 제목
5 * @param {string} content 내용
6 * @param {string} place 장소
7 * @param {string} start 시작
8 * @param {string} end 종료
9 * @param {string} type 종류
10 * @param {string} payment 지불여부
11 */
12 function Festv(num,subject,content,place,start,end,type,payment){
13    this.num = num+1; // 1번부터 번호를 매기기 위해 1증가
14    this.subject = subject;
15    this.content = content;
16    this.place = place;
17    this.start = start;
18    this.end = end;
19    this.type = type;
20    this.payment = payment;
21    this.template = function(){
22        return ` 
23        <div id="card" class="bg-light p-4 rounded-3 border border-primary mb-3">
24            <h3 class="text-truncate" title="${this.subject}">${this.num}-${this.subject}</h3>
25            <div class="my-3">${this.content==null?"내용이없습니다":this.content}</div>
26            <div>${this.place}</div>
27            <div>
28                <span class="badge bg-info">${this.start}</span>
29                <span class="badge bg-secondary">${this.end}</span>
30            </div>
31            <div class="fw-bold">${this.type}</div>
32            <div>${this.payment}</div>
33        </div>`
34    };
35}
